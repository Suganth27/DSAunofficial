#include<bits/stdc++.h>
using namespace std;
void insert(int ht[],int d,int n)
{

 int p=0;
 int hf=d%n;
 int in=(hf+p*p)%n;

 while(ht[in]!=-1)
 {
 if(p==n-1)
 {
 p=-1;
 }
 p++;
 in=(hf+(p*p))%n;
 }
 ht[in]=d;
}
int main()
{
 int n,m;
 cin>>n>>m;
 int a2[m];
 int hashtable[n];
 for(int i=0;i<n;i++)
 hashtable[i]=-1;
 for(int i=0;i<n;i++)
 {

 int num;
 cin>>num;
 insert(hashtable,num,n);
 }

 {
 cin>>a2[i];
 }
